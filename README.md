  url_launcher: ^6.2.1<br>
  http: ^1.1.0<br>
  mobx: ^2.2.1<br>
  flutter_mobx: ^2.2.0<br>
  fluttertoast: 8.0.9<br>

  mockito: ^5.4.3<br>
